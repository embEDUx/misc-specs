setenv kernel_addr_r 10800000
setenv fdt_addr_r 15000000
setenv bootm_low 15000000

setenv fdtfile imx6q-cm-fx6.dtb 
setenv kernelfile zImage

mmc dev ${mmcdev}
fatload mmc ${mmcdev}:1 ${kernel_addr_r} ${kernelfile}
fatload mmc ${mmcdev}:1 ${fdt_addr_r} ${fdtfile}
setenv bootargs "earlyprint console=ttymcx3,115200 root=/dev/mmcblk0p2
rootfstype=ext4 rootwait noinitrd cm_fx6_v4l_msize=128M vmalloc=256M"
bootz ${kernel_addr_r} - ${fdt_addr_r}
