setenv loadaddr 10800000
setenv fdtaddr 11000000
setenv bootm_low 18000000

setenv mmcdev 2
setenv fdtfile imx6q-sbc-fx6m.dtb 
setenv kernel zImage

mmc dev ${mmcdev}
fatload mmc ${mmcdev} ${loadaddr} ${kernel}
fatload mmc ${mmcdev} ${fdtaddr} ${fdtfile}
setenv bootargs "console=ttymxc3,115200 root=/dev/mmcblk0p2 rw rootfstype=ext4
rootwait noinitrd mxcfb0:dev=hdmi,1920x1080M-24@50,if=RGB24"
bootz ${loadaddr} - ${fdtaddr}
